    function formatTime(value, decimal) {
      value = Number.isFinite(value) ? Math.max(0,value) : 0;
      const minutes = Math.floor(value/60);
      const seconds = value%60;
      const tail = decimal ? seconds.toFixed(1).padStart(4,'0') : String(Math.floor(seconds)).padStart(2,'0');
      return String(minutes).padStart(2,'0') + ':' + tail;
    }
    function formatLong(value) {
      value = Math.max(0,Math.floor(value));
      const hours = Math.floor(value/3600);
      const minutes = Math.ceil((value%3600)/60);
      return hours ? hours + 'h ' + minutes + 'm' : minutes + ' min';
    }
    function parseTimeInput(value) {
      const text = String(value || '').trim().replace(',', '.');
      if (!text) return NaN;
      const parts = text.split(':').map(Number);
      if (parts.some(part => !Number.isFinite(part))) return NaN;
      if (parts.length === 1) return parts[0];
      if (parts.length === 2) return parts[0]*60+parts[1];
      return parts[0]*3600+parts[1]*60+parts[2];
    }

    function setClipFromInput(side,input) {

      if (!Number.isFinite(value)) {
        input.classList.add('invalid');
        return updateTimeline();

        if (clipA >= clipB) clipB = Math.min(duration || clipA+previousLength,clipA+previousLength);
        if (clipB <= clipA) clipA = Math.max(0,clipB-.1);
      } else {
        clipB = Math.max(.1,Math.min(value,duration || value));
        if (clipB <= clipA) clipA = Math.max(0,clipB-previousLength);
        if (clipB <= clipA) clipB = Math.min(duration || clipA+.1,clipA+.1);
      }

      updateTimeline();
    }

    async function loadStampArchive() {
      const seen = new Set();
      curatedStamps.forEach(stamp => seen.add(decodeURIComponent(stamp.img).toLowerCase()));
      const titles = [];
      for (const category of commonsCategories) {
        if (titles.length >= 100) break;
        try {
          const url = new URL('https://commons.wikimedia.org/w/api.php');
          url.search = new URLSearchParams({
            action:'query',list:'categorymembers',cmtitle:category,cmtype:'file',
            cmlimit:'100',format:'json',origin:'*'
          }).toString();
          const data = await fetch(url).then(response => response.ok ? response.json() : Promise.reject(new Error('archive')));
          const members = data.query && data.query.categorymembers ? data.query.categorymembers : [];
          for (const item of members) {
            const file = item.title.replace(/^File:/,'');
            if (!/\.(jpe?g|png|webp|tiff?)$/i.test(file)) continue;
            const fileYear = file.match(/\b(18\d{2}|19\d{2}|20\d{2})\b/);
            if (!fileYear || Number(fileYear[0]) > 1945) continue;
            const key = file.toLowerCase();
            if (seen.has(key) || /cover|postcard|forger|booklet/i.test(file)) continue;
            seen.add(key);
            titles.push({file:file,category:category});
            if (titles.length >= 100) break;
          }
        } catch (error) {}
      }
      titles.slice(0,100).forEach((entry,itemIndex) => {
        const index = 10+itemIndex;
        if (index >= TOTAL_STAMPS) return;
        const clean = entry.file
          .replace(/\.[^.]+$/,'')
          .replace(/[_-]+/g,' ')
          .replace(/\b(?:MiNr|B00\d|pm|mt)\b/gi,' ')
          .replace(/\s+/g,' ')
          .trim();
        const yearMatch = clean.match(/\b(18|19|20)\d{2}\b/);
        stamps[index] = {
          year:yearMatch ? yearMatch[0] : '—',
          title:clean.slice(0,70) || 'Archivio ' + String(index+1),
          img:'https://commons.wikimedia.org/wiki/Special:Redirect/file/' + encodeURIComponent(entry.file).replace(/%2F/g,'/'),
          it:I18N.it.stampGeneric,
          en:I18N.en.stampGeneric,
          source:entry.category.replace('Category:','').replace(/_/g,' '),
          loading:false
        };
      });
      renderAlbum(true);
    }

    function projectPoint(longitude,latitude) {
      return [(longitude-6.1)/13.1*390,(47.35-latitude)/11.7*520];
    }
    function ringPath(ring) {
      return ring.map((point,index) => {
        const projected = projectPoint(point[0],point[1]);
        return (index ? 'L' : 'M') + projected[0].toFixed(2) + ',' + projected[1].toFixed(2);
      }).join('') + 'Z';
    }
    function topoArc(topology,index) {
      const reverse = index < 0;
      const arc = topology.arcs[reverse ? ~index : index];
      let x = 0;
      let y = 0;
      const scale = topology.transform.scale;
      const translate = topology.transform.translate;
      const points = arc.map(delta => {
        x += delta[0];
        y += delta[1];
        return [x*scale[0]+translate[0],y*scale[1]+translate[1]];
      });
      return reverse ? points.reverse() : points;
    }
    function topoRing(topology,ids) {
      const output = [];
      ids.forEach((id,index) => {
        const arc = topoArc(topology,id);
        if (index) arc.shift();
        output.push.apply(output,arc);
      });
      return output;
    }
    function topoPath(topology,geometry) {
      const polygons = geometry.type === 'Polygon' ? [geometry.arcs] : geometry.arcs;
      return polygons.map(polygon => polygon.map(ring => ringPath(topoRing(topology,ring))).join('')).join('');
    }

        card.style.setProperty('--i',index);

        image.alt = item.title;
        image.loading = 'lazy';

        localized(title,item.title,item.titleEn);
